23def pedir_lista_floats(mensagem):
24 """
25 Solicita ao utilizador uma lista de números decimais.
26
27 A função apresenta uma mensagem ao utilizador e espera que este
28 insira uma sequência de números separados por vírgula. Cada valor
29 é convertido para float.
30
31 :param mensagem: Texto a apresentar ao utilizador para solicitar a entrada
32 :type mensagem: str
33 :return: Lista de números decimais inseridos pelo utilizador
34 :rtype: list[float]
35
36 :raise ValueError: Se algum dos valores inseridos não puder ser
37 convertido para float
38
39 exemplo
40
41 >>> pedir_lista_floats("Digite valores: ")
42 Digite valores: 1.5,2.7,3.9
43 [1.5, 2.7, 3.9]
44 """
45 valores = input(mensagem)
46 lista = []
47 for valor in valores.split(","):
48 lista.append(float(valor))
49 return lista
